from collections import deque


def while_powerset():
    stack = deque()
    stack.append((0, ''))

    while stack:
        idx, subset = stack.popleft()
        if idx == N:

            stack.append((idx + 1, subset + '1'))
            stack.append((idx + 1, subset + '0'))
