```python
import sys

sys.stdin = open('1231.txt')


# alphabet
# left_child
# right_child

def inorder(node):
    if left_child[node]:
        inorder(left_child[node])
    print(alphabet[node], end='')
    if right_child[node]:
        inorder((right_child[node]))


for t in range(10):
    N = int(input())
    alphabet = [''] * (N + 1)
    left_child = [0] * (N + 1)
    right_child = [0] * (N + 1)
    parent = [0] * (N + 1)
    for n in range(N):
        tmp = list(input().split())
        alphabet[int(tmp[0])] = tmp[1]
        if int(tmp[0]) * 2 <= N:
            left_child[int(tmp[0])] = int(tmp[2])
            parent[int(tmp[2])] = int(tmp[0])
            if int(tmp[0]) * 2 + 1 <= N:
                right_child[int(tmp[0])] = int(tmp[3])
                parent[int(tmp[3])] = int(tmp[0])

    # 우선 루트를 찾아야 할 것 같다
    root = 0
    for idx in range(1, N + 1):
        if not parent[idx]:
            root = idx
            break
    # 루트부터 중위순회
    print(f'#{t + 1}', end=' ')
    inorder(root)
    print()
```
